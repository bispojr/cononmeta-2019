\section{Como utilizar a IpC?}
\begin{frame}{Como utilizar a IpC?}
\begin{block}{Sete passos}
	\begin{itemize}[label=(1)]
		\item[(1)] Identifique o objetivo e os conceitos associados;
		\item[(2)] Prepare aulas expositivas a partir dos conceitos;
		\item[(3)] Escolha o material de referência;
		\item[(4)] Elabore as questões conceituais a partir dos objetivos;
		\item[(5)] Escolha a tecnologia de coleta de votos;
		\item[(6)] Aplique em sala de aula e avalie o impacto; e
		\item[(7)] Divulgue a IpC e estimule seus colegas.
	\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Passo 1: Identifique o objetivo e os conceitos associados}
\begin{block}{Identifique o objetivo de aprendizagem}
	\begin{itemize}[label=$\bullet$]
		\item O que os alunos devem aprender?
		\item Quais os propósitos desta aula?
		\item A que servirá, em termos concretos, a minha prática docente?
	\end{itemize}
\end{block} \pause
\begin{center}
	\includegraphics[width=.7\textheight]{images/objetivo.png}
\end{center}
\end{frame}

\begin{frame}{Passo 1: Identifique o objetivo e os conceitos associados}
\begin{block}{Identifique os conteúdos associados aos objetivos}
	\begin{itemize}[label=$\bullet$]
		\item Quais conceitos estão associados a cada objetivo?
		\item O que o aluno deve apropriar-se para que o objetivo seja alcançado?
	\end{itemize}
\end{block} \pause
\begin{block}{Exemplo: Identificar grandezas mensuráveis no contexto diário}
	Medida de
	\begin{itemize}[label=$\bullet$]
		\item comprimento, 
		\item massa, e 
		\item capacidade
	\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Passo 6: Aplique em sala de aula e avalie o impacto}
\begin{block}{Aspectos importantes}
	\begin{itemize}[label=$\bullet$]
		\item ``Sinta'' a dinâmica da sua turma. \pause
		\item Meça a quantidade de questões adequada para a sua turma. \pause
		\item Tenha cuidado no uso das respostas da IpC para fins de pontuação. \pause
		\item Crie sempre um ambiente favorável à aprendizagem, e não o oposto. \pause
		\item Planeje o momento da votação. \pause
		\item Lembre-se que você está sempre em evolução na sua prática docente.
	\end{itemize}
\end{block} 
\end{frame}
